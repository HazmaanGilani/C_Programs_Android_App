/*C Program to write data to File*/ 

#include <stdio.h>
#include<stdlib.h>
int main()
{
   char ch[100];
   FILE *fptr;

   if ((fptr=fopen("program.txt","r"))==NULL){
       printf("Error! opening file");
       exit(0);         
   }

   
   printf("Enter a sentence:\n",ch);
   gets(ch);
   fprintf(fptr,"%s",ch);
   fclose(fptr);
   
   return 0;
}
